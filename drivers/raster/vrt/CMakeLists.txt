set(TARGET_NAME ras_vrt)

add_definitions(-DFRMT_vrt)
set(FORMAT_DEFINITIONS ${FORMAT_DEFINITIONS} "-DFRMT_vrt" PARENT_SCOPE)

include_directories(${CMAKE_CURRENT_SOURCE_DIR})
include_directories(${CMAKE_SOURCE_DIR}/drivers/raster/raw)

set(LIB_HHEADERS
    vrtdataset.h
    gdal_vrt.h
)

set(LIB_HHEADERS_PRIV
)

set(LIB_CSOURCES
    vrtdriver.cpp
    vrtpansharpened.cpp
    vrtrawrasterband.cpp
    vrtsources.cpp
    vrtwarped.cpp
    vrtdataset.cpp
    vrtderivedrasterband.cpp
    vrtfilters.cpp
    vrtrasterband.cpp
    vrtsourcedrasterband.cpp
    pixelfunctions.cpp
    # vrtmultidim.cpp
)

add_library(${TARGET_NAME} OBJECT ${LIB_HHEADERS} ${LIB_HHEADERS_PRIV} ${LIB_CSOURCES})

# build target
set(TARGET_OBJECTS ${TARGET_OBJECTS} $<TARGET_OBJECTS:${TARGET_NAME}> PARENT_SCOPE)

# install headers
if(NOT SKIP_INSTALL_HEADERS AND NOT SKIP_INSTALL_ALL )
    install(FILES ${LIB_HHEADERS} DESTINATION ${INSTALL_INC_DIR} COMPONENT headers)
endif()
